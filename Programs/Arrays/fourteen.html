// Function to print an array
            void printArray(int arr[], int n) {
                for (int i = 0; i < n; i++) {
                    printf("%d ", arr[i]);
                }
                printf("\n");
            }
            
            // Recursive function to generate and print all combinations
            void generateCombinations(int arr[], int data[], int start, int end, int index, int r) {
                if (index == r) {
                    printArray(data, r);
                    return;
                }
            
                for (int i = start; i <= end && end - i + 1 >= r - index; i++) {
                    data[index] = arr[i];
                    generateCombinations(arr, data, i + 1, end, index + 1, r);
                }
            }
            
            int main() {
                int n, r;
            
                printf("Enter the number of elements in the array: ");
                scanf("%d", &n);
            
                int arr[n];
            
                printf("Enter the elements of the array:\n");
                for (int i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
            
                printf("Enter the size of combinations (r): ");
                scanf("%d", &r);
            
                if (r > n) {
                    printf("Invalid input: r should be less than or equal to the number of elements.\n");
                    return 1;
                }
            
                int data[r];
                generateCombinations(arr, data, 0, n - 1, 0, r);
            
                return 0;
            }
